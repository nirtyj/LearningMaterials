Operations

🐍 Python
from collections import deque

q = deque()

# ── Enqueue / Dequeue ──
q.append(1)                  # right → deque([1])
q.append(2)                  # right → deque([1,2])
q.appendleft(0)              # left  → deque([0,1,2])
q.popleft()                  # dequeue → 0 (O(1)!)
q.pop()                      # from right → 2

# ── Peek ──
q[0]                         # front
q[-1]                        # back
☕ Java
Queue<Integer> q = new LinkedList<>();
// or: Deque<Integer> q = new ArrayDeque<>();

// ── Enqueue / Dequeue ──
q.offer(1);                  // add → [1]
q.offer(2);                  // add → [1,2]
// q.offerFirst(0);          // Deque only
q.poll();                    // dequeue → 1
// q.pollLast();             // Deque only

// ── Peek ──
q.peek();                    // front (null if empty)
// ((Deque)q).peekLast();    // Deque only

🛠️ Built-ins commonly paired with queues / deques

When you write queue / deque code in interviews, these are the built-ins you reach for most.

from collections import deque

# ── Building ──
q = deque()
q = deque([root])                         # init with single element (BFS start)
q = deque(arr)                            # init from any iterable
q = deque(maxlen=k)                       # bounded — auto-pops oldest

# ── BFS level-snapshot pattern (the universal trick) ──
for _ in range(len(q)):                   # snapshot size BEFORE iterating
    node = q.popleft()
    q.extend(node.children)               # bulk append — extend = append-many

# ── Iterating without consuming ──
for x in q: ...                          # left-to-right
list(q)                                   # snapshot copy
enumerate(q)                              # index + value

# ── Index-style peek ──
q[0]                                       # front
q[-1]                                      # back

# ── Sliding window with maxlen ──
window = deque(maxlen=k)
for x in stream:
    window.append(x)                      # window auto-trims to last k items
    if len(window) == k: print(sum(window))

Interview Pattern: BFS (Level-Order / Shortest Path)

Queue-based level traversal for trees / graphs. Used in: Level Order Traversal, Rotting Oranges, Word Ladder, Walls and Gates, Number of Islands

🐍 Python Shortcuts
# ── Template: Tree level-order BFS ──
from collections import deque
def levelOrder(root):
    if not root: return []
    res, q = [], deque([root])
    while q:
        level = []
        for _ in range(len(q)):         # ← process one level
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

# ── Template: Graph/Grid BFS ──
def bfs(grid, start):
    q = deque([start])
    visited = {start}                    # ← set for O(1) lookup
    steps = 0
    while q:
        for _ in range(len(q)):
            r, c = q.popleft()           # ← tuple unpacking
            for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:  # ← 4 directions
                nr, nc = r + dr, c + dc
                if 0 <= nr < rows and 0 <= nc < cols \
                   and (nr, nc) not in visited:
                    visited.add((nr, nc)) # ← tuple as set element
                    q.append((nr, nc))
        steps += 1
    return steps

# ── Python tricks used ──
# • deque([start])      — O(1) popleft (list.pop(0) is O(n)!)
# • for _ in range(len(q))  — snapshot level size
# • (r, c) tuples       — hashable, can go in sets
# • 0 <= nr < rows      — chained comparison
