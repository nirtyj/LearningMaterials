Operations

🐍 Python
import heapq

# ── Min Heap ──
heap = []
heapq.heappush(heap, 5)      # [5]
heapq.heappush(heap, 1)      # [1,5]
heapq.heappush(heap, 3)      # [1,5,3]
heap[0]                      # peek min → 1
heapq.heappop(heap)          # pop min → 1

# ── Max Heap (negate values!) ──
heapq.heappush(heap, -val)
-heapq.heappop(heap)

# ── Heapify existing list ──
nums = [5, 3, 8, 1]
heapq.heapify(nums)          # O(n), in-place

# ── Push + Pop combined ──
heapq.heappushpop(heap, val) # push then pop min
heapq.heapreplace(heap, val) # pop min then push

# ── Top K Pattern ──
heapq.nlargest(3, nums)      # 3 largest
heapq.nsmallest(3, nums)     # 3 smallest

# ── With tuples (priority, data) ──
heapq.heappush(heap, (2, "low"))
heapq.heappush(heap, (1, "high"))
heapq.heappop(heap)          # (1, "high")
☕ Java
// ── Min Heap ──
PriorityQueue<Integer> pq = new PriorityQueue<>();
pq.offer(5);                 // [5]
pq.offer(1);                 // [1,5]
pq.offer(3);                 // [1,5,3]
pq.peek();                   // peek min → 1
pq.poll();                   // pop min → 1

// ── Max Heap (pass comparator!) ──
PriorityQueue<Integer> maxPQ =
    new PriorityQueue<>(Collections.reverseOrder());

// ── Heapify existing collection ──
PriorityQueue<Integer> pq =
    new PriorityQueue<>(Arrays.asList(5,3,8,1));

// ── No push+pop combo ──
// Must do manually: pq.offer(val); pq.poll();

// ── No built-in nlargest/nsmallest ──
// Use stream: nums.stream().sorted()
//   .limit(3).toList();

// ── With custom objects ──
PriorityQueue<int[]> pq = new PriorityQueue<>(
    (a,b) -> a[0] - b[0]);     // sort by first elem
pq.offer(new int[]{2, 0});  // {priority, data}
pq.offer(new int[]{1, 1});
pq.poll();                   // [1, 1]

🛠️ Built-ins commonly paired with heaps

When you write heap code in interviews, these are the built-ins you reach for most.

import heapq

# ── Top-K shortcuts (bypass heappush/heappop loops) ──
heapq.nlargest(k, arr)                    # top k as list
heapq.nsmallest(k, arr)
heapq.nlargest(k, items, key=lambda x: x.score)  # objects by attribute

# ── Tuples as priorities ──
heapq.heappush(pq, (priority, tiebreaker, payload))
val, _, obj = heapq.heappop(pq)           # tuple unpacking on pop

# ── Heapify in place (O(n) — faster than n × heappush) ──
heapq.heapify(arr)

# ── Counter + heapq combo (top-k frequent) ──
from collections import Counter
count = Counter(arr)
heapq.nlargest(k, count, key=count.get)   # keys with top-k counts

# ── Iterating: a heap is just a list ──
for x in pq: ...                         # NOT in sorted order — array layout
for x in sorted(pq): ...                 # sorted snapshot (allocates)
len(pq), pq[0]                          # size, peek-min — no method call needed

# ── Max-heap via negation ──
heapq.heappush(pq, -val)
-heapq.heappop(pq)                        # flip sign back on extract

Full reference for any of these: Python Builtins.

Interview Pattern: Heap / Top-K / Two Heaps

Priority queue for K-th element, streaming median. Used in: Kth Largest Element, Top K Frequent, Merge K Sorted Lists, Find Median from Data Stream, Task Scheduler

🐍 Python Shortcuts
import heapq

# ── Top K Frequent Elements (LC 347) ──
def topKFrequent(nums, k):
    count = Counter(nums)                # ← instant frequency
    return heapq.nlargest(k, count, key=count.get)  # ← one-liner!
    # Alt: return [x for x, _ in count.most_common(k)]

# ── Kth Largest Element (LC 215) ──
def findKthLargest(nums, k):
    return heapq.nlargest(k, nums)[-1]   # ← built-in!

# ── Merge K Sorted Lists (LC 23) ──
def mergeKLists(lists):
    heap = []
    for i, l in enumerate(lists):
        if l:
            heapq.heappush(heap, (l.val, i, l))  # ← i breaks ties!
    dummy = cur = ListNode(0)
    while heap:
        val, i, node = heapq.heappop(heap)
        cur.next = node
        cur = cur.next
        if node.next:
            heapq.heappush(heap, (node.next.val, i, node.next))
    return dummy.next

# ── Find Median (LC 295) — Two Heaps ──
# small = max-heap (negate!), large = min-heap
small, large = [], []  # max-heap (neg), min-heap
heapq.heappush(small, -num)
heapq.heappush(large, -heapq.heappop(small))

# ── Python tricks used ──
# • heapq.nlargest(k, ...)  — avoid full sort
# • Counter().most_common(k) — top K frequencies
# • -val for max-heap       — negate trick
# • (val, tiebreaker, obj)  — tuple comparison order
