üí° Interview Insight: Mutability

  • Python: Lists are mutable, tuples/strings are immutable
  • Java: Arrays are mutable, Strings are immutable
  • In Python, a = b for lists creates a reference, not a copy!
  • Use a = b.copy() or a = b[:] for shallow copy

üí° Key Initialization Gotchas

  • {} creates an empty dict in Python, NOT a set. Use set() for empty sets
  • [[0]*n]*m creates m references to the SAME inner list ‚Äî use a list comprehension instead
  • Java's List.of(), Map.of(), Set.of() create immutable collections ‚Äî wrap in new ArrayList<>(List.of(...)) if you need mutable
  • Python's defaultdict is one of the biggest time-savers ‚Äî no more key-existence checks before inserting

def build_tree(vals):

            q.append(node.left)
        i += 1
        if i < len(vals) and vals[i] is not None:

        i += 1
    return root

def threeSum(nums):
    nums.sort()
    res = []
    for i in range(len(nums) - 2):
        if i > 0 and nums[i] == nums[i-1]:
            continue                     # ‚Üê skip duplicates
        l, r = i + 1, len(nums) - 1
        while l < r:
            s = nums[i] + nums[l] + nums[r]
            if s < 0: l += 1
            elif s > 0: r -= 1
            else:

                while l < r and nums[l] == nums[l+1]: l += 1
                l += 1; r -= 1
    return res

from collections import deque
def levelOrder(root):
    if not root: return []
    res, q = [], deque([root])
    while q:
        level = []
        for _ in range(len(q)):         # ‚Üê process one level
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

def bfs(grid, start):

    steps = 0
    while q:
        for _ in range(len(q)):

            for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:  # ‚Üê 4 directions

                   and (nr, nc) not in visited:
                    visited.add((nr, nc)) # ‚Üê tuple as set element
                    q.append((nr, nc))
        steps += 1
    return steps

        res.append(path[:])              # ‚Üê copy! (path[:] or list(path))

def permute(nums):
    res = []

        if len(path) == len(nums):
            res.append(path[:])
            return
        for n in nums:
            if n not in path:            # ‚Üê O(n) but clean for interviews
                path.append(n)
                backtrack(path)
                path.pop()
    backtrack([])
    return res

def combinationSum(candidates, target):
    res = []
    def backtrack(start, path, total):
        if total == target:
            res.append(path[:])
            return
        for i in range(start, len(candidates)):
            if total + candidates[i] > target:
                break                    # ‚Üê prune (sort first!)
            path.append(candidates[i])
            backtrack(i, path, total + candidates[i])  # i not i+1 (reuse)
            path.pop()
    candidates.sort()                    # ‚Üê sort enables pruning
    backtrack(0, [], 0)
    return res

def coinChange(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for a in range(1, amount + 1):
        for c in coins:
            if c <= a:
                dp[a] = min(dp[a], dp[a - c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1

def longestCommonSubsequence(s1, s2):
    dp = [[0] * (len(s2)+1) for _ in range(len(s1)+1)]
    for i in range(1, len(s1)+1):
        for j in range(1, len(s2)+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[-1][-1]                    # ‚Üê dp[-1][-1] = bottom-right

def canFinish(numCourses, prerequisites):
    graph = defaultdict(list)
    indegree = [0] * numCourses
    for course, pre in prerequisites:
        graph[pre].append(course)
        indegree[course] += 1
    q = deque(i for i in range(numCourses) if indegree[i] == 0)
    taken = 0                            # ‚Üê generator in deque!
    while q:
        node = q.popleft()
        taken += 1
        for nei in graph[node]:
            indegree[nei] -= 1
            if indegree[nei] == 0:
                q.append(nei)
    return taken == numCourses

# ‚îÄ‚îÄ Topological Sort ‚Äî DFS cycle detection ‚îÄ‚îÄ
def canFinish_dfs(numCourses, prerequisites):
    graph = defaultdict(list)
    for c, p in prerequisites: graph[p].append(c)
    WHITE, GRAY, BLACK = 0, 1, 2       # ‚Üê state enum
    color = [WHITE] * numCourses
    def dfs(node):
        color[node] = GRAY
        for nei in graph[node]:
            if color[nei] == GRAY: return False  # cycle!
            if color[nei] == WHITE and not dfs(nei): return False
        color[node] = BLACK
        return True
    return all(dfs(i) for i in range(numCourses) if color[i] == WHITE)

# ‚îÄ‚îÄ Template: Union Find with path compression + rank ‚îÄ‚îÄ
class UnionFind:
    def __init__(self, n):
        self.par = list(range(n))       # ‚Üê parent[i] = i initially
        self.rank = [0] * n

    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])  # ‚Üê path compression
        return self.par[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py: return False         # already connected
        if self.rank[px] < self.rank[py]:
            px, py = py, px              # ‚Üê swap trick
        self.par[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        return True
