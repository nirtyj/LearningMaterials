Mutable Default Arguments

# ❌ WRONG - list persists between calls!
def add(item, lst=[]):
    lst.append(item)
    return lst

# ✅ CORRECT
def add(item, lst=None):
    if lst is None:
        lst = []
    lst.append(item)
    return lst

Integer Division

# Python 3
7 / 2   # = 3.5 (float division)
7 // 2  # = 3   (integer division)

# Negative numbers — // FLOORS toward -∞ (not toward 0)
-7 // 2             # = -4   (floors toward -∞)
int(-7 / 2)         # = -3   (truncate toward 0, via float)
-(-7 // 2) * -1     # = -3   (pure-int truncate; ugly but no float)

# Ceiling division — two idioms, both pure-int:
-(-7 // 3)            # = 3    (negate-floor-negate)
math.ceil(7 / 3)        # = 3    (clearer; uses float)

Modulo with Negatives

# Python's % follows the SIGN OF THE DIVISOR (always ≥ 0 for positive b)
(-7) % 3             # = 2   (because -7 = -3*3 + 2)
7 % (-3)             # = -2  (matches sign of divisor)

# For truncate-toward-0 behavior (returns float):
import math
math.fmod(-7, 3)        # = -1.0

# Why this matters: hashing, "wrap-around" indices, and
# circular-array problems — Python's behavior is usually what you want
(idx - 1) % n          # wraps -1 to n-1 cleanly

List Copy vs Reference

# ❌ Reference (same object)
a = [1, 2, 3]
b = a
b.append(4)  # a is also [1,2,3,4]!

# ✅ Shallow copy
b = a[:]       # or a.copy()
b = list(a)

# ✅ Deep copy (nested structures)
import copy
b = copy.deepcopy(a)
