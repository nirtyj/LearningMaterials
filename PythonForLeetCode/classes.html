📦 LeetCode Node Definitions

These are pre-defined on LeetCode — you don't write them, but you must know their shape.

# ── Binary Tree Node ──
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

# ── Singly Linked List Node ──
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

# ── N-ary Tree Node ──
class Node:
    def __init__(self, val=None, children=None):
        self.val = val
        self.children = children or []  # ← avoid mutable default!

# ── Doubly Linked List Node (LRU Cache) ──
class DLLNode:
    def __init__(self, key=0, val=0):
        self.key = key
        self.val = val
        self.prev = None
        self.next = None

# ── Graph Node (Clone Graph) ──
class Node:
    def __init__(self, val=0, neighbors=None):
        self.val = val
        self.neighbors = neighbors or []

# ── Trie Node ──
class TrieNode:
    def __init__(self):
        self.children = {}             # char → TrieNode
        self.is_end = False

# ⚡ Shortcut: Trie as nested dicts (no class!)
trie = {}
for c in word:
    trie = trie.setdefault(c, {})

🔑 Key Python Differences for Nodes

  • No type declarations on fields — just assign in __init__
  • Default args: val=0, left=None — creates node in one call: TreeNode(5)
  • Mutable default trap: Use children=None + or [], never children=[]
  • Python nodes are not typed — node.val can be anything (int, str, tuple)
  • No access modifiers (no private / public) — everything is accessible

🔨 Constructing Nodes Inline

🐍 Python — Inline Construction
# ── Build a linked list: 1 → 2 → 3 → 4 → 5 ──
head = ListNode(1, ListNode(2, ListNode(3, ListNode(4, ListNode(5)))))

# ── Build from a list ──
def build_list(arr):
    dummy = ListNode(0)
    cur = dummy
    for val in arr:
        cur.next = ListNode(val)
        cur = cur.next
    return dummy.next

head = build_list([1, 2, 3, 4, 5])

# ── Build a binary tree inline ──
#       1
#      / \
#     2   3
#    / \
#   4   5
root = TreeNode(1,
    TreeNode(2, TreeNode(4), TreeNode(5)),
    TreeNode(3)
)

# ── Build a binary tree from list (LeetCode-style) ──
from collections import deque
def build_tree(vals):
    if not vals: return None
    root = TreeNode(vals[0])
    q = deque([root])
    i = 1
    while q and i < len(vals):
        node = q.popleft()
        if i < len(vals) and vals[i] is not None:
            node.left = TreeNode(vals[i])
            q.append(node.left)
        i += 1
        if i < len(vals) and vals[i] is not None:
            node.right = TreeNode(vals[i])
            q.append(node.right)
        i += 1
    return root

root = build_tree([1, 2, 3, None, None, 4, 5])

# ── Build graph adjacency list ──
from collections import defaultdict
graph = defaultdict(list)
edges = [[0,1], [0,2], [1,3]]
for u, v in edges:
    graph[u].append(v)
    graph[v].append(u)                  # ← undirected

🐍 Python Class Building Blocks

The OOP toolkit you'll need for design questions.

# ── Basic Class ──
class Animal:
    def __init__(self, name, sound):
        self.name = name               # public (no modifier)
        self._sound = sound             # "private" by convention

    def speak(self):
        return f"{self.name} says {self._sound}"

    def __repr__(self):                # string form (for debug/print)
        return f"Animal('{self.name}')"

    def __eq__(self, other):           # equality (==)
        return self.name == other.name

    def __hash__(self):               # enables use as dict key / set element
        return hash(self.name)

    def __lt__(self, other):           # ordering — enables sorted() and heapq
        return self.name < other.name

# ── Inheritance ──
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "Woof")
        self.breed = breed

    def speak(self):                   # override
        return f"{self.name} barks!"

# ── Multiple Inheritance (no interface!) ──
class Flyable:
    def fly(self): return "flying"

class Swimmable:
    def swim(self): return "swimming"

class Duck(Animal, Flyable, Swimmable):
    pass                              # gets all methods!

📋 @dataclass — Auto-Generate Boilerplate

Python 3.7+. Auto-generates __init__, __repr__, and __eq__ from declared fields. Perfect for OOD interviews.

from dataclasses import dataclass, field

# ── Basic: auto-generates __init__, __repr__, __eq__ ──
@dataclass
class Point:
    x: int
    y: int

p = Point(3, 4)                       # __init__ is free!
print(p)                               # Point(x=3, y=4)
Point(3, 4) == Point(3, 4)             # True (__eq__ is free!)

# ── With ordering (enables sorted / heapq!) ──
@dataclass(order=True)
class Task:
    priority: int                       # ← sorted by first field!
    name: str

tasks = [Task(3, "low"), Task(1, "high"), Task(2, "mid")]
sorted(tasks)                           # [Task(1,'high'), Task(2,'mid'), Task(3,'low')]
# Also works with heapq!
import heapq
heapq.heappush(heap, Task(1, "urgent"))  # ← compares by priority

# ── Frozen (immutable + hashable → usable as dict key/set) ──
@dataclass(frozen=True)
class Coordinate:
    x: int
    y: int

visited = set()
visited.add(Coordinate(1, 2))         # ← hashable! works in sets

# ── Mutable default fields (use field!) ──
@dataclass
class Graph:
    name: str
    edges: list = field(default_factory=list)  # ← NOT edges=[] !

# ── Post-init validation ──
@dataclass
class Employee:
    name: str
    age: int
    def __post_init__(self):
        if self.age < 0:
            raise ValueError("Age can't be negative")

📎 NamedTuple — Lightweight Immutable Records

🐍 Python NamedTuple
from typing import NamedTuple

# ── Clean syntax (Python 3.6+) ──
class Point(NamedTuple):
    x: int
    y: int

p = Point(3, 4)
p.x                                    # 3 (dot access)
x, y = p                               # unpacking!
p[0]                                   # 3 (index access)
hash(p)                                # hashable (works in sets/dicts!)

# ── Great for: multi-value returns, dict keys, BFS states ──
class State(NamedTuple):
    row: int
    col: int
    keys: frozenset                    # ← immutable set for hashing

visited = set()
visited.add(State(0, 0, frozenset())) # ← works as set element!

# ── When to use which? ──
# NamedTuple → immutable, hashable, can be dict key
# @dataclass  → mutable by default, more features
# @dataclass(frozen=True) → immutable like NamedTuple but slower
# Plain tuple → quick and dirty, no named fields

⚖️ Custom Sorting & Comparators

Sorting objects, heap ordering, custom comparators — key= functions, __lt__, and cmp_to_key.

# ── Sort with key function (most common) ──
intervals.sort(key=lambda x: x[0])    # by first elem
intervals.sort(key=lambda x: (x[0], -x[1]))  # multi-key!
words.sort(key=len)                    # by length
words.sort(key=str.lower)              # case-insensitive

# ── Sort objects ──
people.sort(key=lambda p: p.age)
people.sort(key=lambda p: (-p.score, p.name))  # desc score, asc name

# ── @total_ordering: define __eq__ + __lt__, get all 6 ──
from functools import total_ordering

@total_ordering
class Student:
    def __init__(self, name, grade):
        self.name = name
        self.grade = grade

    def __eq__(self, other):
        return self.grade == other.grade

    def __lt__(self, other):
        return self.grade < other.grade
    # __le__, __gt__, __ge__ auto-generated!

sorted(students)                        # works!
min(students)                           # works!
heapq.heappush(heap, student)           # works!

# ── functools.cmp_to_key (rare, for complex comparators) ──
from functools import cmp_to_key
def compare(a, b):
    # return negative if a < b, 0 if equal, positive if a > b
    return int(a + b) - int(b + a)     # LC 179: Largest Number

nums_str = sorted(map(str, nums), key=cmp_to_key(compare), reverse=True)

# ── Singleton ──
class Singleton:
    _instance = None
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

# ── Iterator ──
class BSTIterator:                      # LC 173
    def __init__(self, root):
        self.stack = []
        self._push_left(root)

    def _push_left(self, node):
        while node:
            self.stack.append(node)
            node = node.left

    def __next__(self):                # ← makes it a Python iterator
        node = self.stack.pop()
        self._push_left(node.right)
        return node.val

    def __iter__(self):
        return self

# Usage: for val in BSTIterator(root): print(val)

# ── Property (getters/setters the Python way) ──
class Circle:
    def __init__(self, radius):
        self._radius = radius

    @property
    def radius(self):
        return self._radius

    @radius.setter
    def radius(self, value):
        if value < 0: raise ValueError
        self._radius = value

    @property
    def area(self):                    # ← computed property
        return 3.14159 * self._radius ** 2

c = Circle(5)
c.radius                               # 5 (no () needed!)
c.area                                 # 78.54 (computed, no ()!)

# ── Static / Class Methods ──
class MathUtils:
    @staticmethod
    def gcd(a, b):                     # no self — pure utility, callable on class
        while b: a, b = b, a % b
        return a

    @classmethod
    def from_string(cls, s):           # factory method
        return cls(int(s))

# ── OrderedDict (LRU Cache — LC 146) ──
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity):
        self.cache = OrderedDict()
        self.cap = capacity

    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)     # ← built-in method!
        return self.cache[key]

    def put(self, key, val):
        self.cache[key] = val
        self.cache.move_to_end(key)
        if len(self.cache) > self.cap:
            self.cache.popitem(last=False)  # ← remove oldest!

🎯 When to Use What — OOP Quick Decision Guide

  • Plain class — when you need custom methods, mutable state, complex logic (LRU Cache, Trie, etc.)
  • @dataclass — structured data with auto __init__/__eq__/__repr__. Add order=True for sortable, frozen=True for immutable
  • NamedTuple — immutable records, great as dict keys or BFS states with complex keys
  • Enum / IntEnum — named constants, directions, states. IntEnum when you need comparisons
  • @total_ordering — when your object needs to go in a heap or sorted list, define only __eq__ + __lt__
  • ABC — design interview scaffolding, enforcing interfaces
  • @property — computed fields, validation on set; access like an attribute (no parens)
  • Nested dicts — quick Trie or adjacency list without class boilerplate
